
 CREATE TABLE "SYSTEM"."CUSTOMER" 
   (	"CNO" NUMBER NOT NULL ENABLE, 
	"CNAME" VARCHAR2(20 BYTE), 
	"CADD" VARCHAR2(20 BYTE), 
	"BILLAMT" FLOAT(126), 
	 CONSTRAINT "CUSTOMER_PK" PRIMARY KEY ("CNO")
	 
	 
	  CREATE SEQUENCE  "SYSTEM"."CNO_SEQ"  MINVALUE 1 MAXVALUE 1000 INCREMENT BY 1 START WITH 1 CACHE 20 NOORDER  NOCYCLE ;
